def extract_apkg(apkg_path: Path) -> dict:
    """
    –†–∞—Å–ø–∞–∫–æ–≤—ã–≤–∞–µ—Ç .apkg —Ñ–∞–π–ª –∏ –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç —Å—Ç—Ä—É–∫—Ç—É—Ä—É –¥–∞–Ω–Ω—ã—Ö

            'db_path': Path,  # –ü—É—Ç—å –∫ SQLite –±–∞–∑–µ
            'media_json': dict,  # –°–ª–æ–≤–∞—Ä—å –º–µ–¥–∏–∞—Ñ–∞–π–ª–æ–≤ {—á–∏—Å–ª–æ: –∏–º—è_—Ñ–∞–π–ª–∞}

        # –ò–∑–≤–ª–µ–∫–∞–µ–º –≤—Å–µ —Ñ–∞–π–ª—ã –≤–æ –≤—Ä–µ–º–µ–Ω–Ω—É—é –¥–∏—Ä–µ–∫—Ç–æ—Ä–∏—é

        temp_dir.mkdir(parents=True, exist_ok=True)
        
        zip_ref.extractall(temp_dir)
        
        # –ß–∏—Ç–∞–µ–º media JSON
        media_file = temp_dir / 'media'
        media_json = {}
        if media_file.exists():
            with open(media_file, 'r', encoding='utf-8') as f:
                media_json = json.load(f)
        
        # –ß–∏—Ç–∞–µ–º –º–µ–¥–∏–∞—Ñ–∞–π–ª—ã
        media_files = {}
        for num_str, filename in media_json.items():
            media_path = temp_dir / num_str
            if media_path.exists():
                with open(media_path, 'rb') as f:
                    media_files[num_str] = f.read()
        
        return {
            'db_path': temp_dir / 'collection.anki2',
            'media_json': media_json,
            'media_files': media_files,
            'temp_dir': temp_dir
        }

```python
def parse_anki_db(db_path: Path) -> dict:
    """
    –ü–∞—Ä—Å–∏—Ç SQLite –±–∞–∑—É Anki –∏ –∏–∑–≤–ª–µ–∫–∞–µ—Ç –∫–∞—Ä—Ç–æ—á–∫–∏
    
    Returns:
        {
            'deck_name': str,
            'cards': [
                {
                    'fields': [str, ...],  # –ü–æ–ª—è –∫–∞—Ä—Ç–æ—á–∫–∏
                    'original_word': str,  # –ü–µ—Ä–≤–æ–µ –ø–æ–ª–µ
                    'translation': str,    # –í—Ç–æ—Ä–æ–µ –ø–æ–ª–µ
                    'audio': str,          # –ò–º—è –∞—É–¥–∏–æ—Ñ–∞–π–ª–∞ (–µ—Å–ª–∏ –µ—Å—Ç—å)
                    'image': str,          # –ò–º—è —Ñ–∞–π–ª–∞ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è (–µ—Å–ª–∏ –µ—Å—Ç—å)
                },
                ...
            ]
        }
    """
    conn = sqlite3.connect(str(db_path))
    cursor = conn.cursor()
    
    # –ü–æ–ª—É—á–∞–µ–º –Ω–∞–∑–≤–∞–Ω–∏–µ –∫–æ–ª–æ–¥—ã
    cursor.execute("SELECT decks FROM col LIMIT 1")
    decks_json = cursor.fetchone()[0]
    decks = json.loads(decks_json)
    deck_name = list(decks.values())[0]['name'] if decks else '–ò–º–ø–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω–∞—è –∫–æ–ª–æ–¥–∞'
    
    # –ü–æ–ª—É—á–∞–µ–º –∑–∞–ø–∏—Å–∏ (notes)
    cursor.execute("""
        SELECT id, flds, sfld 
        FROM notes
        ORDER BY id
    """)
    
    cards = []
    for note_id, flds, sfld in cursor.fetchall():
        # –ü–æ–ª—è —Ä–∞–∑–¥–µ–ª–µ–Ω—ã —Å–∏–º–≤–æ–ª–æ–º \x1f
        fields = flds.split('\x1f')
        
        # –ò–∑–≤–ª–µ–∫–∞–µ–º –º–µ–¥–∏–∞—Ñ–∞–π–ª—ã –∏–∑ –ø–æ–ª–µ–π
        audio = None
        image = None
        
        for field in fields:
            # –ê—É–¥–∏–æ: [sound:filename.mp3]
            if '[sound:' in field:
                audio = field.split('[sound:')[1].split(']')[0]
            # –ò–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ: <img src="filename.jpg">
            if '<img src="' in field:
                image = field.split('<img src="')[1].split('"')[0]
        
        # –ü–µ—Ä–≤–æ–µ –ø–æ–ª–µ - —Å–ª–æ–≤–æ, –≤—Ç–æ—Ä–æ–µ - –ø–µ—Ä–µ–≤–æ–¥
        original_word = fields[0] if len(fields) > 0 else ''
        translation = fields[1] if len(fields) > 1 else ''
        
        cards.append({
            'fields': fields,
            'original_word': original_word,
            'translation': translation,
            'audio': audio,
            'image': image,
        })
    
    conn.close()
    
    return {
        'deck_name': deck_name,
        'cards': cards
    }
```

def save_media_files(media_json: dict, media_files: dict, user) -> dict:
    """
    –°–æ—Ö—Ä–∞–Ω—è–µ—Ç –º–µ–¥–∏–∞—Ñ–∞–π–ª—ã –≤ MEDIA_ROOT –∏ –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –º–∞–ø–ø–∏–Ω–≥
    
    Returns:
        {
            'audio_filename': '/media/audio/new_filename.mp3',
            'image_filename': '/media/images/new_filename.jpg',
            ...
        }
    """
    media_mapping = {}
    
    for num_str, original_filename in media_json.items():
        if num_str not in media_files:
            continue
        
        file_content = media_files[num_str]
        
        # –û–ø—Ä–µ–¥–µ–ª—è–µ–º —Ç–∏–ø —Ñ–∞–π–ª–∞ –ø–æ —Ä–∞—Å—à–∏—Ä–µ–Ω–∏—é
        if original_filename.endswith(('.mp3', '.ogg', '.wav')):
            # –ê—É–¥–∏–æ—Ñ–∞–π–ª
            new_filename = f"{uuid.uuid4()}.{original_filename.split('.')[-1]}"
            media_path = Path(settings.MEDIA_ROOT) / 'audio' / new_filename
            media_path.parent.mkdir(parents=True, exist_ok=True)
            
            with open(media_path, 'wb') as f:
                f.write(file_content)
            
            media_mapping[original_filename] = f'/media/audio/{new_filename}'
            
        elif original_filename.endswith(('.jpg', '.jpeg', '.png', '.gif', '.webp')):
            # –ò–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ
            new_filename = f"{uuid.uuid4()}.{original_filename.split('.')[-1]}"
            media_path = Path(settings.MEDIA_ROOT) / 'images' / new_filename
            media_path.parent.mkdir(parents=True, exist_ok=True)
            
            with open(media_path, 'wb') as f:
                f.write(file_content)
            
            media_mapping[original_filename] = f'/media/images/{new_filename}'
    
    return media_mapping
```

def create_deck_from_import(user, deck_name: str, cards: list, media_mapping: dict, language: str = 'de'):
    """
    –°–æ–∑–¥–∞–µ—Ç –∫–æ–ª–æ–¥—É –∏ —Å–ª–æ–≤–∞ –∏–∑ –∏–º–ø–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω—ã—Ö –¥–∞–Ω–Ω—ã—Ö
    """
    # –°–æ–∑–¥–∞–µ–º –∫–æ–ª–æ–¥—É
    deck = Deck.objects.create(
        user=user,
        name=deck_name,
        target_lang=language,
        source_lang=user.native_language or 'ru'
    )
    
    # –°–æ–∑–¥–∞–µ–º —Å–ª–æ–≤–∞
    for card_data in cards:
        word_obj, created = Word.objects.get_or_create(
            user=user,
            original_word=card_data['original_word'],
            language=language,
            defaults={
                'translation': card_data['translation'],
            }
        )
        
        # –ü—Ä–∏–≤—è–∑—ã–≤–∞–µ–º –º–µ–¥–∏–∞—Ñ–∞–π–ª—ã
        if card_data['audio'] and card_data['audio'] in media_mapping:
            word_obj.audio_file.name = media_mapping[card_data['audio']].replace('/media/', '')
        
        if card_data['image'] and card_data['image'] in media_mapping:
            word_obj.image_file.name = media_mapping[card_data['image']].replace('/media/', '')
        
        word_obj.save()
        
        # –î–æ–±–∞–≤–ª—è–µ–º —Å–ª–æ–≤–æ –≤ –∫–æ–ª–æ–¥—É
        deck.words.add(word_obj)
    
    return deck
```
